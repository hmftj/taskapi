function longest() { fetch('https://restcountries.com/v3.1/all') .then(function(response) { if (!response.ok) { throw new Error('HTTP error! status: ' + response.status); } return response.json(); }) .then(function(countries) { var longestCountryName = ""; countries.forEach(function(country) { if (country.name.common.length > longestCountryName.length) { longestCountryName = country.name.common; } }); console.log("The country with the longest name is:"); console.log(longestCountryName); }) .catch(function(error) { console.error("Error fetching country data:", error); }); }
